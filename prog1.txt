init R0, 1
ld R1, R0       # R1 = Q

init R3, 3      # R3 = MUX[3] = 6, the result of calculation
init R0, 1      # the exponential counter

L1:
R3x6            # R3 = R3 * 6
inc R0
add R3, R1      # R3 = R3 + Q

jpu R1, R3, L2  # jump to L2 if R1 < R3 (Q < Result)
